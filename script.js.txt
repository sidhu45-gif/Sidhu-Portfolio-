// Typing effect
const nameText = "Sidhu Pottabathini";
const typingEl = document.getElementById("typing-name");
let idx = 0;
function typing() {
if (idx < nameText.length) {
typingEl.innerHTML += nameText.charAt(idx);
idx++;
setTimeout(typing, 120);
}
}
typing();
// Certificate flip logic with auto flip back
let currentTimer = null;
let lastFlipped = null;
document.querySelectorAll(".cert-card").forEach(function(card) {
card.addEventListener("click", function() {
if (!card.classList.contains("flip")) {
if (lastFlipped && lastFlipped !== card) {
lastFlipped.classList.remove("flip");
}
card.classList.add("flip");
lastFlipped = card;
if (currentTimer) clearTimeout(currentTimer);
currentTimer = setTimeout(function() {
card.classList.remove("flip");
lastFlipped = null;
currentTimer = null;
}, 5000);
}
});
});
});

const menuBtn = document.getElementById("menu-toggle");
const menuNav = document.getElementById("side-menu");
function closeMenu() {
menuNav.classList.add("hidden");
document.body.classList.remove("menu-active");
}
function openMenu() {
menuNav.classList.remove("hidden");
document.body.classList.add("menu-active");
}
menuBtn.addEventListener("click", openMenu);
document.addEventListener("keydown", function(e) {
if (!menuNav.classList.contains("hidden") && (e.key === "Escape" || e.key === "Esc")) {
closeMenu();
}
});
window.navigateTo = function(id) {
closeMenu();
setTimeout(function() {
document.getElementById(id).scrollIntoView({ behavior: "smooth" });
}, 220);
}
